# This file goes in this folder /etc/nginx/sites-available/territorytools.org
server {
    server_name territorytools.org;

    location / {
        proxy_pass          http://localhost:6000/;
        proxy_set_header    Host                $http_host;
        proxy_set_header    X-Real-IP           $remote_addr;
        proxy_set_header    X-Forwarded-For     $proxy_add_x_forwarded_for;
        proxy_set_header    X-Forwarded-Host    $host;
        proxy_set_header    X-Forwarded-Proto   $scheme;
    }


    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/territorytools.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/territorytools.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = territorytools.org) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    server_name territorytools.org;
    listen 80;
    return 404; # managed by Certbot


}
